import java.util.*;

public class Solution {
  
    public static int[] getProductArrayExceptSelf(int[] arr) {
    	//Your code goes here
        
        int size = arr.length;
        // Answer Array having  size same as product array
        int ans[] = new int[size];
        
        for (int i =0;i<size;i++)
        {
            //Sending current element 
            //System.out.println("Array Element is :"+arr[i]);
             ans[i]=ProdSum(arr,i); 
            //System.out.println("Answer Element is :"+ans[i]);
        }
        
        //Return the Product Array
        return ans;
    }
    
    private static int ProdSum(int[] arr,int currentElement){
        //System.out.println("Current element is :"+currentElement);
        int p=1;
        for(int i =0;i<arr.length;i++){
          //Problem is passing of array :Always it should be 5 2 2  
            
          System.out.println("Array Element is :"+arr[i]);
            if (i==currentElement){
               //Setting current value as 1 
               arr[i]=1;
               //Flushing Activity needs to be done 
            }
            
            p=p*arr[i];
            //System.out.println("Product return is :"+p);
        }
        
       //It did not flush value : arr=null;
        return p;
    }
     
}
